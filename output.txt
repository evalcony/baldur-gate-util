// 法师战斗脚本
IF
HPPercentLT(Myself,50)
!GlobalTimerNotExpired("random_walk","LOCALS")
THEN
RESPONSE #100
ReallyForceSpell(Myself,WIZARD_IMPROVED_INVISIBILITY)
SetGlobalTimer("random_walk","LOCALS",1)
Continue()
END

IF
GlobalTimerNotExpired("random_walk","LOCALS")
THEN
RESPONSE #100
DisplayString(Myself,1000)
RESPONSE #12
MoveToOffset([-30.0])
RESPONSE #12
MoveToOffset([0.30])
RESPONSE #12
MoveToOffset([30.0])
RESPONSE #12
MoveToOffset([0.-30])
RESPONSE #12
MoveToOffset([30.30])
RESPONSE #12
MoveToOffset([-30.30])
RESPONSE #12
MoveToOffset([30.-30])
RESPONSE #12

END
